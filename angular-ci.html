export DISPLAY=:1
Xvfb :1 &

curl -s -o use-node https://repository-cloudbees.forge.cloudbees.com/distributions/ci-addons/node/use-node
NODE_VERSION=0.10.26 . ./use-node

npm install
npm install karma-junit-reporter

# Start web server for testing
npm start > /dev/null &
NODE_PID=$!

npm run update-webdriver
sleep 1 # give server time to start

# Now run the end to end tests
node_modules/.bin/karma start test/karma.conf.js --no-auto-watch --single-run --reporters="dots,junit" --browsers=Firefox
node_modules/.bin/protractor test/protractor-conf.js --browser=firefox --reporters="dots,junit"


# Cleanup
kill -s TERM $NODE_PID
